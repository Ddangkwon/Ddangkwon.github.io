# 격자 안에서 완전탐색  최고의 33위치

단순 완전 탐색, 브루트포스 문제이다.

```cpp
#include <cstdio>

using namespace std;

int N;
int arr[20][20] = { 0 };

int main() {
    // 여기에 코드를 작성해주세요.
    int max_cnt = 0;
    scanf("%d", &N);
    for(int i = 0; i < N; i++)
    {
        for(int j = 0; j < N; j++)
        {
            scanf("%d", &arr[i][j]);   
        }
    }

    
    for(int i = 0; i < N; i++)
    {
        for(int j = 0; j < N; j++)
        {
            int temp_cnt = 0;
            for(int y = i; y < i + 3; y++)
            {
                for(int x = j; x < j + 3; x++)
                {
                    if(arr[y][x] == 1)
                        temp_cnt++;

                }
            }
            if(max_cnt < temp_cnt)
                max_cnt = temp_cnt;
        }
    }
    printf("%d\n", max_cnt);

    return 0;
}
```
