# 정수 사각형 최장 증가 수열

[https://www.codetree.ai/ko/trails/complete/curated-cards/challenge-lis-on-the-integer-grid/description](https://www.codetree.ai/ko/trails/complete/curated-cards/challenge-lis-on-the-integer-grid/description)


## 풀이

backtracking 으로 문제 해결

```cpp
#include <iostream>
#include <algorithm>
using namespace std;

int n;
int grid[500][500];
int dp[500][500];
int answer = 0;

int dy[4] = { -1, 1, 0, 0 };
int dx[4] = { 0, 0, -1, 1 };


int dfs(int y, int x){
    int max_cnt = 1;
    if(dp[y][x] != -1)
        return dp[y][x];

    for(int dir = 0; dir < 4; dir++){
        int ny = y + dy[dir];
        int nx = x + dx[dir];

        if(ny < 0 || ny >= n || nx < 0 || nx >= n)
            continue;
        if(grid[ny][nx] > grid[y][x])
            max_cnt = max(max_cnt, dfs(ny, nx) + 1);
    }


    dp[y][x] = max_cnt;
    return dp[y][x];
}
int main() {

    cin >> n;

    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            cin >> grid[i][j];
            dp[i][j] = -1;
        }
    }

    
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            answer = max(answer, dfs(i, j));
        }
    }

    cout << answer << "\n";

    return 0;
}
```
