# 배열 돌리기 2

첫 번째 배열 돌리기 문제에서 돌리는 횟수가 크게 증가할 경우 복잡성을 낮추는 문제이다.
우리가 돌리려고 하는 사각형들의 총 길이를 알 수 있다면 그 숫자의 나머지만큼만 회전시키면 되기 때문에 1과 조금 다르게 rotate 함수에서 
몇 번 회전시켜줄지를 결정한다.


## 내 풀이
```cpp
#include <iostream>
#include <algorithm>


using namespace std;

int arr[301][301] = { {0, }, };
int N, M, R;

// 우 하 좌 상
int dy[4] = { 0,1,0,-1 };
int dx[4] = { 1,0,-1,0 };
void input_func()
{
    scanf("%d %d %d", &N, &M, &R);

    for (int i = 0; i < N; i++) {
        for (int j = 0; j < M; j++) {
            scanf("%d", &arr[i][j]);
        }
    }
}
void rotate(int startp,int len) {
    int rect_len = R % len;  //회전 횟수 설정
    for (int i = 0; i < rect_len; i++) 
    {   
       
        int startPoint = arr[startp][startp]; // 박스 시작 포인트 설정(시작 지점 백업)
        int cy = startp;
        int cx = startp;

        int k = 0;
        while (k < 4) {

            int ny = cy + dy[k];    
            int nx = cx + dx[k];
            //시작 지점은 그대로 둔다
            if (ny == startp && nx == startp) 
                break;
            if (startp <= ny && ny < N - startp && startp <= nx && nx < M - startp) 
            {

                //차례로 시계 반대방향으로 옮김
                arr[cy][cx] = arr[ny][nx];
                cy = ny;
                cx = nx;

            }
            else {       //다음에 옮길 칸이 배열 범위 넘어가버리면 해당 라인은 다 옮긴거라서 k 증가
                k++;
            }
        }
        arr[startp + 1][startp] = startPoint; //백업 데이터 저장

    }


}
void print_output()
{
    //결과 출력
    for (int i = 0; i < N; i++)
    {
        for (int j = 0; j < M; j++)
        {
            printf("%d ", arr[i][j]);
        }
        printf("\n");
    }
}
int main() 
{

    input_func();
    

    int cnt = min(N, M) / 2;       //  박스 수
    int n = N, m = M;

    for (int i = 0; i < cnt; i++) 
    {       //박스 수만큼 회전 동작 반복
        int len = (2 * n) + (2 * m) - 4;
        rotate(i, len);
        n -= 2;
        m -= 2;

    }


    print_output();



    return 0;
}
```
