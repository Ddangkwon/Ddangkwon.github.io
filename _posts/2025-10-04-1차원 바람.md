# 1차원 바람

[https://www.codetree.ai/ko/trails/complete/curated-cards/challenge-The-1D-wind-blows/description](https://www.codetree.ai/ko/trails/complete/curated-cards/challenge-The-1D-wind-blows/description)



## 풀이

```cpp
#include <iostream>

using namespace std;

int n, m, q;
int arrmap[101][101];

void input(){
    cin >> n >> m >> q;

    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= m; j++) {
            cin >> arrmap[i][j];
        }
    }
}

void shift_right(int row){
    int temp_val = arrmap[row][m]; 
    for(int i = m; i > 1; i--){
        arrmap[row][i] = arrmap[row][i - 1];
    }
    arrmap[row][1] = temp_val;
}

void shift_left(int row){
    int temp_val = arrmap[row][1]; 
    for(int i = 1; i < m; i++){
        arrmap[row][i] = arrmap[row][i + 1];
    }
    arrmap[row][m] = temp_val;
}

void print_answer(){
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= m; j++) {
            cout << arrmap[i][j] << " ";
        }
        cout << "\n";
    }
}

void simulate(int r, int d){
    int cur_pos = r, uppos = r - 1, downpos = r + 1, updir = d, downdir = d;
    bool up_flag = true, down_flag = true;
    if(d == 0){
        shift_right(r);
        updir = 1;
        downdir = 1;
    }
    else{
        shift_left(r);
        updir = 0;
        downdir = 0;
    }
    while(uppos >= 1){
        bool temp_flag = false;
        for(int i = 1; i <= m; i++){
            if(arrmap[cur_pos][i] == arrmap[uppos][i])
                temp_flag = true;
        }  
        if(!temp_flag)
            break;
        cur_pos--;
        uppos--;
        if(updir == 0){
            shift_right(cur_pos);
            updir = 1;
        }
        else{
            shift_left(cur_pos);
            updir = 0;
        }
    }
    // current pos 초기화
    cur_pos = r; 
    while(downpos <= n){
        bool temp_flag = false;
        for(int i = 1; i <= m; i++){
            if(arrmap[cur_pos][i] == arrmap[downpos][i])
                temp_flag = true;
        }  
        if(!temp_flag)
            break;
        cur_pos++;
        downpos++;
        if(downdir == 0){
            shift_right(cur_pos);
            downdir = 1;
        }
        else{
            shift_left(cur_pos);
            downdir = 0;
        }
    }

}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(NULL);
    cout.tie(NULL);

    input();

    for (int i = 0; i < q; i++) {
        int r, dir;
        char d;
        cin >> r >> d;
        // d == L, 왼쪽에서 민다는 의미로 실제 이동은 오른쪽 쉬프트
        if (d == 'L'){
            dir = 0;
        }
        // d == R, 오른쪽에서 민다는 의미로 실제 이동은 왼쪽 쉬프트
        else{
            dir = 1;
        }
        simulate(r, dir);
    }


    print_answer();

    return 0;
}
```
