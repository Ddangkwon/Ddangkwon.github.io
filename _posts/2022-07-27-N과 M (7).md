# N과 M (6) 
순열 문제로 재귀와 백트래킹을 사용하여 문제를 해결한다.

중복되는 순열을 출력 가능 => 방문 여부 체크 X

```cpp
#include <iostream>
#include <cstdio>
#include <algorithm>

using namespace std;
int N, M;
int arr[9] = { 0, };
int ansarr[9] = { 0, };
void dfs(int cnt)
{
	if (cnt == M)
	{
		for (int i = 0; i < M; i++)
		{
			printf("%d ", ansarr[i]);
		}
		printf("\n");
		return;
	}

	for (int i = 0; i < N; i++)
	{
		ansarr[cnt] = arr[i];
		dfs(cnt + 1);
	}
}

int main()
{
	scanf("%d %d", &N, &M);
	for (int i = 0; i < N; i++)
	{
		scanf("%d", &arr[i]);
	}
	sort(arr, arr + N);
	dfs(0);

	return 0;
}
```
