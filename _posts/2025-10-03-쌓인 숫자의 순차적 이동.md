# 쌓인 숫자의 순차적 이동

[https://www.codetree.ai/ko/trails/complete/curated-cards/challenge-sequential-movement-of-stacked-numbers/description](https://www.codetree.ai/ko/trails/complete/curated-cards/challenge-sequential-movement-of-stacked-numbers/description)



## 풀이

```cpp
#include <iostream>
#include <deque>

using namespace std;

int n, m;
deque<int> grid[20][20];
int move_nums[100];

int dy[8] = { -1, -1, 0, 1, 1, 1, 0, -1 };
int dx[8] = { 0, 1, 1, 1, 0, -1, -1, -1 };


void input() {
    cin >> n >> m;

    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            int num;
            cin >> num;
            grid[i][j].push_back(num);
        }
    }

    for (int i = 0; i < m; i++) {
        cin >> move_nums[i];
    }
}

void simulate() {

    for (int i = 0; i < m; i++) {

        int search_num = move_nums[i];
        int cy = 0, cx = 0, cidx = 0;
        bool find_flag = false;
        // 현재 업데이트할 숫자 좌표 탐색
        for (int r = 0; (r < n && !find_flag); r++) {
            for (int c = 0; (c < n && !find_flag); c++) {

                for (int j = 0; j < grid[r][c].size(); j++) {
                    if (grid[r][c][j] == search_num) {
                        find_flag = true;
                        cy = r;
                        cx = c;
                        cidx = j;
                        break;
                    }
                }

                if (find_flag)
                    break;
            }
            if (find_flag)
                break;
        }

        // 매칭되는 숫자가있는 좌표를 찾을 경우 8방향 최대값 탐색
        int max_num = 0, mov_y = cy, mov_x = cx;
        if (find_flag) {
            for (int dir = 0; dir < 8; dir++) {
                int ny = cy + dy[dir];
                int nx = cx + dx[dir];

                if (ny < 0 || ny >= n || nx < 0 || nx >= n) {
                    continue;
                }

                for (int j = 0; j < grid[ny][nx].size(); j++) {
                    if (max_num < grid[ny][nx][j]) {
                        max_num = grid[ny][nx][j];
                        mov_y = ny;
                        mov_x = nx;
                    }
                }

            }

            // 최종 탐색한 위치로 값 이동
            
            for (int j = cidx; j >= 0; j--) {
                grid[mov_y][mov_x].push_front(grid[cy][cx][j]);
            }
            // 이동한 값은 원래 좌표에서 삭제
            cidx += 1;
            grid[cy][cx].erase(grid[cy][cx].begin(), grid[cy][cx].begin() + cidx);

        }
    }
}
int main() {

    ios::sync_with_stdio(false);
    cin.tie(NULL);
    cout.tie(NULL);
    input();

    simulate();

    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            if (!grid[i][j].empty()) {
                for (int k = 0; k < grid[i][j].size(); k++) {
                    cout << grid[i][j][k] << " ";
                }
                cout << "\n";
            }
            else {
                cout << "None\n";
            }
        }
    }
    return 0;
}
```
