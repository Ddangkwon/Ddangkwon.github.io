### dijkstra 알고리즘 개념 정리

```cpp
// priority_queue 함수 functor 를 정의, priority_queue 는 기본적으로 최대힙이기 때문에 최소힙으로 표현하기 위해 이를 사용함
struct compare{
    bool operator()(const data_t& a, const data_t& b) const{
        return a.time > b.time;
    }
};

void dijkstra(int start)
{
    priority_queue<data_t, vector<data_t>, compare> pq;

    // 초기화 시퀀스, 최소 값을 찾아야 할 때는 이렇듯 탐색 배열을 최대값으로 초기화 해야함
    for(int i = 0; i <= n; i++){
        dist[i] = INT_MAX;
    }
    pq.push({start, 0});
    dist[start] = 0;

    while(!pq.empty()){
        data_t cur_data = pq.top();
        int cur_dest = cur_data.dest;
        int cur_time = cur_data.time;

        pq.pop();
        // PQ 에서 꺼낸 정보가 현재 dist 배열에 기록된 최신 최단거리 정보와 다르면, 그 항목은 과거에 삽입된 낡은 항목이므로 무시
        if(dist[cur_dest] != cur_time) // dist[cur_dest] < cur_time
            continue;
        // 현재 노드의 업데이트가 완료 됐을 경우 넘어감, 주로 시간 단축이나, 탐색 노드의 개수를 count 하기 위해 사용
        // 동일 최소값으로 여러번 노드가 업데이트 될 수 있기 때문에 아래와 같은 done flag 사용, 아니면 위 if 문을 != 가 아닌 < 로 표현 해야함.
        if(done[cur_dest])
            continue;
        done[cur_dest] = true;
        // 위 if 문을 통해 Dijkstra 과정에서 최단거리가 확정된 정점 개수를 카운트
        def_comp_cnt++;
        // 최대값 탐색 필요한 경우
        last_comp_def_time = max(last_comp_def_time, dist[cur_dest]);
        for(auto& next_data : graph[cur_dest]){
            int next_dest = next_data.dest;
            int next_time = dist[cur_dest] + next_data.time;

            if(next_time < dist[next_dest]){
                pq.push({next_dest, next_time});
                dist[next_dest] = next_time;
            }
        }
    }
}
```
