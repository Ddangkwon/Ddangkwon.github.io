# 격자 안에서 완전탐색 트로미노

테트리스 처럼 특정 블럭을 회전시켜가며 최대 합을 구하는 문제이다.
배열의 회전 등을 경험해 볼 수 있는 좋은 문제.

```cpp
#include <cstdio>

using namespace std;

int n, m;
int arr[200][200] = { 0 };
int dy[4] = {-1, 0, 1, 0};
int dx[4] = {0, 1, 0, -1};
int main()
{
    int max_cnt = 0;
    scanf("%d %d", &n , &m);

    for(int i = 0; i < n; i++)
    {
        for(int j = 0; j < m; j++)
        {
            scanf("%d", &arr[i][j]);
        }
    }

    for(int i = 0; i < n; i++)
    {
        for(int j = 0; j < m; j++)
        {
            int flag = 1;
            int temp_cnt = 0;
            for(int idx = 0; idx < 4; idx++)
            {

                flag = 1;
                temp_cnt = arr[i][j];
                int y = i + dy[idx];
                int x = j + dx[idx];
                // 테트리스 블럭이 범위 내에 들어가는지 확인한다(flag)
                if(y >= 0 && y < n && x >= 0 && x < m)
                {
                    temp_cnt += arr[y][x];
                }
                else
                {
                    flag = 0;
                }

                y = i + dy[(idx + 1) % 4];
                x = j + dx[(idx + 1) % 4];
                if(y >= 0 && y < n && x >= 0 && x < m)
                {
                    temp_cnt += arr[y][x];
                }
                else
                {
                    flag = 0;
                }  
                if(max_cnt < temp_cnt && flag == 1)
                {                    
                    max_cnt = temp_cnt;
                }
            }
            
            for(int idx = 0; idx < 4; idx++)
            {
                flag = 1;
                temp_cnt = arr[i][j];
                int y = i + dy[idx];
                int x = j + dx[idx];
                if(y >= 0 && y < n && x >= 0 && x < m)
                {
                    temp_cnt += arr[y][x];
                }
                else
                {
                    flag = 0;
                }  
                y = i + dy[(idx + 2) % 4];
                x = j + dx[(idx + 2) % 4];
                if(y >= 0 && y < n && x >= 0 && x < m)
                {
                    temp_cnt += arr[y][x];
                }
                else
                {
                    flag = 0;
                }
                if(max_cnt < temp_cnt && flag == 1)
                {                    
                    max_cnt = temp_cnt;
                }
            }
        }
    }
    printf("%d", max_cnt);
    return 0;
}
```
