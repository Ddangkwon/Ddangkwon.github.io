# 로프
그리디 문제로, 병렬로 균등하게 무게가 배분되므로 로프의 하중 견디는 값이 높은 로프를 쓸수록 로프는 적게 쓸 수 밖에 없어진다.
이를 생각하며 정렬을 사용하여 문제를 해결한다.
```cpp
#include <iostream>
#include <cstdio>
#include <algorithm>

using namespace std;
#define MAX_NUM 100000

int N;
int weight[MAX_NUM] = { 0 };
int rope[MAX_NUM] = { 0 };
int main(void)
{
	scanf("%d", &N);
	for (int i = 0; i < N; i++)
	{
		scanf("%d", &rope[i]);
	}
	sort(rope, rope + N, greater<int>());
	for (int i = 0; i < N; i++)
	{
		weight[i] = rope[i]  * (i + 1);
	}
	sort(weight, weight + N, greater<int>());
	cout << weight[0] << '\n';
	return 0;
}

```
