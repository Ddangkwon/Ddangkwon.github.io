# 집합의 표현



## 풀이

```cpp
#include <iostream>

#define MAX_N 1000001
using namespace std;

int n, m;
int parent[MAX_N] = { 0 };
int rankv[MAX_N] = { 0 };

int find_parent(int x)
{
    if(x == parent[x])
        return x;

    return parent[x] = find_parent(parent[x]);
}

void union_parent(int x, int y)
{
    int px = find_parent(x);
    int py = find_parent(y);

    if(px == py)
        return ;

    if(rankv[px] < rankv[py])
        parent[px] = py;
    else{
        parent[py] = px;
        if(rankv[px] == rankv[py])
            rankv[px]++;
    }
}
int check_parent_same(int x, int y)
{
    return find_parent(x) == find_parent(y);
}

int main(void)
{
    ios::sync_with_stdio(false);
    cin.tie(NULL);
    cout.tie(NULL);

    cin >> n >> m;

    for(int i = 1; i <= n; i++)
        parent[i] = i;
    for(int i = 0; i < m; i++){
        int cmd, a, b;
        cin >> cmd >> a >> b;

        if(cmd){
            int flag = check_parent_same(a, b);
            if(flag)
                cout << "YES\n";
            else
                cout << "NO\n";
        }
        else
            union_parent(a, b);

    }
    return 0;
}
```
