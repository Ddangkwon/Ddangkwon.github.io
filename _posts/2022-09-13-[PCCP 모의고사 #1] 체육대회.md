# [PCCP 모의고사] 체육대회


```cpp

#include <string>
#include <vector>
#include <algorithm>
using namespace std;
bool isvisited[10] = {false, };
int answer = 0, tempsum = 0;
// call by reference
void dfs(vector<vector<int>> &ability, int idx)
{
    if(idx == ability[0].size())
    {
        if(answer < tempsum)
        {
            answer = tempsum;
        }
        return;
    }
    // 조합 구하기 => 중복 방지용 방문 체크 수행
    for(int i = 0; i < ability.size(); i++)
    {
        if(isvisited[i] == 1)
            continue;
        isvisited[i] = true;
        tempsum += ability[i][idx];
        dfs(ability, idx + 1);
        tempsum -= ability[i][idx];
        isvisited[i] = false;
    }
    return;
}
int solution(vector<vector<int>> ability) {
    
    dfs(ability, 0);
    return answer;
}
```
