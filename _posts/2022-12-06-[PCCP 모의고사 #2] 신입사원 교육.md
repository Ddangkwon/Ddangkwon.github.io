# 신입사원 교육

단순 sort 방식으로는 시간 초과 발생
```cpp
#include <string>
#include <vector>
#include <algorithm>
using namespace std;

int solution(vector<int> ability, int number) {
    int answer = 0, min_sum = 0;
    for(int i = 0; i < number; i++)
    {
        sort(ability.begin(), ability.end());
        min_sum = ability[0] + ability[1];
        ability[0] = min_sum;
        ability[1] = min_sum;
    }
    for(int i = 0; i < ability.size(); i++)
    {
        answer += ability[i];
    }
    return answer;
}
```

우선순위 큐로 최소힙 구현 후 문제 해결
```cpp
#include <string>
#include <vector>
#include <queue>
using namespace std;

int solution(vector<int> ability, int number) {
    priority_queue<int, vector<int>, greater<int>> pq;
    int answer = 0, min_sum = 0;
    for(int i = 0; i < ability.size(); i++)
    {
        pq.push(ability[i]);   
    }
    for(int i = 0; i < number; i++)
    {
        int a, b;
        a = pq.top();
        pq.pop();
        b =  pq.top();
        pq.pop();
        min_sum = a + b;
        pq.push(min_sum);
        pq.push(min_sum);
        
    }
    while(!pq.empty())
    {
        answer += pq.top();
        pq.pop();
    }
    return answer;
}
```
