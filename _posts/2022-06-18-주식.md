# 주식
그리디 문제로, 뒤의 날짜 즉 미래를 보면서 수익 계산을 하면 용이하다.
뒤에서부터 최대 이익을 계산하면서 최고가를 보면 max_price값을 정정하면서 수익을 집계한다.

(저가) -> (연중 최고가, 매도) -> (저가) -> (저가) -> (주중 최고가, 매도)
```cpp
#include <iostream>
#include <cstdio>
#include <algorithm>

using namespace std;
typedef long long ll;

int T, N;
int price[1000000] = { 0 };

int main(void)
{
	
	scanf("%d", &T);
	for (int i = 0; i < T; i++)
	{
		ll max_benefit = 0;
		scanf("%d", &N);
		for(int j = 1; j <= N; j++)
		{
			scanf("%d", &price[j]);
		}
		int max_price = price[N];

		for(int j = N - 1; j > 0; j--)
		{
			if (max_price <= price[j])
			{
				max_price = price[j];
			}
			else
			{
				max_benefit += (ll)(max_price - price[j]);
			}
		}
		cout << max_benefit << '\n';
	}
	

	return 0;
}

```
