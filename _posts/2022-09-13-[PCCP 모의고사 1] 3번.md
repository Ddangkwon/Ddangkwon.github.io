# [PCCP 모의고사 1] 3번

유전 형질 0, 1, 1, 2 패턴을 파악하고 재귀로 문제를 해결한다.


```cpp
#include <string>
#include <vector>
#include <cmath>
#include <iostream>
using namespace std;
int dfs(int n, int p)
{
    if(n == 2)
    {
        // 2세대일 경우 별도 처리해준다.
        if(p == 1)
        {
            return 0; 
        }
        else if(p == 4)
        {
            return 2;
        }
        else
        {
            return 1;    
        }        

    }
    
    int s = p / 4;
    int r = p % 4;
    // 버림처리한 숫자를 다시 올림 처리 해준다.
    if(r != 0)
    {
        s++;
    }
    int share = dfs(n - 1, s);
    // 맨 첫단은 무조건 RR이 맨 끝단은 무조건 rr이 나타난다.
    if(share == 0)
    {
        return 0;
    }
    else if(share == 2)
    {
        return 2;
    }
    else
    {
        if(r == 1)
        {
            return 0;
        }
        
        else if(r == 0)
        {
            return 2;
        }
        else
        {
            return 1;
        }
    }

}
vector<string> solution(vector<vector<int>> queries) {
    vector<string> answer;
    
    for(int i = 0; i < queries.size(); i++)
    {
        int temp_answer = 0;
        int n = queries[i][0], p = queries[i][1];

        if(n == 1)
        {
            answer.push_back("Rr");
            continue;
        }

        temp_answer = dfs(n, p);      
        
        
        if(temp_answer == 0)
        {
            answer.push_back("RR");
        }
        else if(temp_answer == 1)
        {
            answer.push_back("Rr");
        }
        else
        {
            answer.push_back("rr");
        }
            
    }
    return answer;
}
```
