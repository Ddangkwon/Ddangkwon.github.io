오목 승패를 판단하는 문제

```cpp
#include <cstdio>
#include <iostream>
#include <vector>
#include <algorithm>

#define MAX_ARR_LEN 20

using namespace std;

int arr[MAX_ARR_LEN][MAX_ARR_LEN] = { 0 };
// int dy[8] = { -1, -1, -1, 0, 1, 1, 1, 0 };
// int dx[8] = { -1, 0, 1, 1, 1, 0, -1, -1 };

// 왼쪽 우선 그 다음 아래 우선이므로 방향은
// 우상, 우, 우하, 하  경로만 탐색해야한다.
int dy[4] = { -1, 0, 1, 1};
int dx[4] = {1, 1, 1, 0};
int main()
{
    int answer_cnt = 0, winner = 0, pos_y, pos_x;
    for(int i = 1; i <= 19; i++){
        for(int j = 1; j <= 19; j++){
            scanf("%2d", &arr[i][j]);
        }
    }

    for(int i = 1; i <= 19; i++){
        for(int j = 1; j <= 19; j++){
            if(arr[i][j] == 1 || arr[i][j] == 2){
                int temp = arr[i][j];
                for(int dir = 0; dir < 4; dir++){
                    answer_cnt = 1;
                    int ny = i + dy[dir];
                    int nx = j + dx[dir];
                    // printf("=======start============\n");
                    // printf("%d %d\n", i, j);
                    // 다음 루트가 범위를 벗어날 경우 다시 해볼 필요 없음
                    if(!(ny >= 1 && ny <= 19 && nx >=1 && nx <=19))
                        continue;
                    // 두번째도 같은 숫자인지 체크
                    if(arr[ny][nx] != temp)
                        continue;
                    // 이미 지나온길 하나만 체크
                    // 육목인데 부분적으로 판단해서 오목으로 볼 수 있으므로 
                    int oy = i - dy[dir];
                    int ox = j - dx[dir];
                    if((oy >= 1 && oy <= 19 && ox >=1 && ox <=19 ))
                        if(arr[oy][ox] == temp)
                        continue;
                    // do while 문이므로 결국 세번째부터 오목인지 판단하는 코드
                    do {
                        
                        if(!(ny >= 1 && ny <= 19 && nx >=1 && nx <=19 ))
                            continue;
                        answer_cnt++;
                        // printf("%d %d current_cnt : %d target : %d\n", ny, nx, answer_cnt, temp);
                        ny = i + (answer_cnt * dy[dir]);
                        nx = j + (answer_cnt * dx[dir]);
                         
                    }while(arr[ny][nx] == temp);
                    
                    if(answer_cnt == 5){
                        winner = temp;
                        pos_y = i;
                        pos_x = j;
                        break;
                    }
                }
            }
            if(winner != 0)
                break; 
        }
        if(winner != 0)
            break;

    }


    printf("%d\n", winner);
    if(winner != 0)
        printf("%d %d\n", pos_y, pos_x);




    return 0;
}
```

```cpp
#ifndef _CRT_SECURE_NO_WARNINGS
#define _CRT_SECURE_NO_WARNINGS
#endif

#include <stdio.h>

int T, N, K;
int A[10];

/*
* 중복 조합
* A[c]를 선택해
*/
void recur1(int c) {
    // base case
    if (c > N) {
        for (int i = 1; i <= N; i++)
            printf("%d ", A[i]);
        printf("\n");
        return;
    }

    // normal case
    // 조건을 만족하는 것 부터, 즉 앞 숫자와 같은 숫자가 다음에 올 수 있다.
    for (int i = A[c-1]; i <= K; i++) {
        A[c] = i;
        recur1(c + 1);
    }
}

/* 조합 */
void recur2(int c) {
    if (c > N) {
        for (int i = 1; i <= N; i++)
            printf("%d ", A[i]);
        printf("\n");
        return;
    }
    // 중복을 허용하지 않아야하므로 앞에 숫자보다 더 커야하고 따라서 +1 을 해주고 A[0] 은 0으로 해준다.
    for (int i = A[c-1] + 1; i <= K; i++) {
        A[c] = i;
        recur2(c + 1);
    }
}

int main() {
#ifndef ONLINE_JUDGE
    freopen("input.txt", "r", stdin);
#endif
```

```cpp
#include <iostream>
#include <cstdio>
#include <vector>
#include <queue>

using namespace std;

# define MAX_ARR_LEN  102
int n, m;
int arr[MAX_ARR_LEN][MAX_ARR_LEN] = { 0 };
int time_arr[MAX_ARR_LEN][MAX_ARR_LEN] = { 0 };
bool isvisited[MAX_ARR_LEN][MAX_ARR_LEN] = { false };
int dy[4] = {-1, 0, 1, 0};
int dx[4] = {0, 1, 0, -1};
queue<pair<int, int>> q;
void bfs(int y, int x)
{
    q.push(make_pair(y, x));
    isvisited[y][x] = true;
    time_arr[y][x] = 3;
    while(!q.empty())
    {
        int cy = q.front().first;
        int cx = q.front().second;
        q.pop();

        for(int dir = 0; dir < 4 ; dir++){
            int ny = cy + dy[dir];
            int nx = cx + dx[dir];
            if(ny >= 1 && ny <= m && nx >= 1 && nx <= n){
                if(arr[ny][nx] == 1 && !isvisited[ny][nx])
                {
                    isvisited[ny][nx] = true;
                    q.push(make_pair(ny, nx));
                    
                    time_arr[ny][nx] = time_arr[cy][cx] + 1;
                }
            }

        }

    }

}
int main()
{
    //  1.입력부
    int zerg_cnt = 0;
    scanf("%d %d", &n, &m);
    for(int i = 1; i <= m; i++){
        for(int j = 1; j <= n; j++){
            scanf("%1d", &arr[i][j]);
        }
    }
    int y, x;
    scanf("%d %d", &x, &y);
    // bfs 탐색 실행
    bfs(y, x);
    // 최대 시간 탐색
    int temp_max = 0;
    for(int i = 1; i <= m; i++){
        for(int j = 1; j <= n; j++){
            if(temp_max < time_arr[i][j])
                temp_max = time_arr[i][j];
        }
    }

    // 남은 저글링수 탐색
    printf("%d\n", temp_max);
    for(int i = 1; i <= m; i++){
        for(int j = 1; j <= n; j++){
            if(arr[i][j] == 1 && !isvisited[i][j])
                zerg_cnt++;
        }
```

```cpp
                     q.push(make_pair(ny, nx));
                     isvisited[ny][nx] = true;
                }
                
            }
        }
    }
    group.push_back(data_cnt);
}   
int main()
{
    scanf("%d", &n);

    // 1. 입력부
    for(int i = 1; i <= n; i++){
        for(int j = 1; j <= n; j++){
            scanf("%1d", &arr[i][j]);
        }
    }

    // 2. bfs 실행 및 정렬
    
    for(int i = 1; i <= n; i++){
        for(int j = 1; j <= n; j++){
            if(arr[i][j] == 1 && !isvisited[i][j]){
                bfs(i, j);
                group_cnt++;
            }
        }
    }
    sort(group.begin(), group.end());
    // 3. 출력부
    printf("%d\n", group_cnt);
    
    for(int i = 0; i < group.size(); i++)
        printf("%d\n", group[i]);

    return 0;
}
```

### bfs 최단거리 문제 (체스 말로 위치 고정된 폰 잡기)

```cpp
#include <cstdio>
#include <iostream>
#include <queue>
#include <vector>

#define MAX_NUM 1100
using namespace std;

int N, M, R, C, S, K;
int dy[8] = {-2, -2, -1, 1, 2, 2, 1, -1};
int dx[8] = {-1, 1, 2, 2, 1, -1, -2, -2};
int arr[MAX_NUM][MAX_NUM] = { 0 };
int move_cnt[MAX_NUM][MAX_NUM] = { 0 };
bool isvisited[MAX_NUM][MAX_NUM] = { false };
queue<pair<int, int> > q;

void bfs(int y, int x)
{
    q.push(make_pair(y, x));
    isvisited[y][x] = true;

    while(!q.empty()){
        int cy = q.front().first;
        int cx = q.front().second;
        
        q.pop();
        for(int dir = 0; dir < 8; dir++){
            int ny = cy + dy[dir];
            int nx = cx + dx[dir];

            if(ny >= 1 && ny <= N && nx >= 1 && nx <= M){
                if(!isvisited[ny][nx]){
                    q.push(make_pair(ny, nx));
                    isvisited[ny][nx] = true;
                    // 이전보다 하나를 더 더한값이 최신상태에서는 최단거리 
                    move_cnt[ny][nx] = move_cnt[cy][cx] + 1;
                }
            }
        }
        // 폰을 잡으면 탈출
        if(isvisited[S][K]){
            printf("%d\n", move_cnt[S][K]);
            break;
        }
            
    }
}

int main(){
    scanf("%d %d", &N, &M);

    scanf("%d %d %d %d", &R, &C, &S, &K);

    bfs(R, C);
    return 0;
}
```
