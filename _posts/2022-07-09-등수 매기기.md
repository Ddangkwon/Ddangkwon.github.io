# 등수 매기기
그리디 문제로, 동석차 없이 1~N까지 숫자를 줄세울 수 있도록 만드는 문제
=> 정렬하고 오차 발생시 절댓값의 오차 만큼을 정답 카운트에 더해줌

```cpp
#include <iostream>
#include <cstdio>
#include <vector>
#include <algorithm>
using namespace std;
int N;
vector<int> v;
int main(void)
{
	long long answer = 0;
	scanf("%d", &N);

	for (int i = 0; i < N; i++)
	{
		int num;
		scanf("%d", &num);
		v.push_back(num);
	}
	sort(v.begin(), v.end());
	for (int i = 1; i <= N; i++)
	{
		answer += abs(i - v[i - 1]);
	}
	cout << answer << '\n';
	return 0;
}

```
