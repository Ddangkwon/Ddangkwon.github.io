#include <iostream>
#include <cstdio>
#include <algorithm>

int N;
int point[100] = { 0, };

int main(void)
{
	int answer = 0;
	scanf("%d", &N);
	for (int i = 0; i < N; i++)
	{
		scanf("%d", &point[i]);
	}

	for (int i = N - 1; i > 0; i--)
	{
		if (N == 1)
			break;

		while (point[i - 1] >= point[i])
		{
			point[i - 1]--;
			answer++;

		}
	}

	cout << answer << '\n';
	return 0;
}
