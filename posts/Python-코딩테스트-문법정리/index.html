def dfs(li,V,visited):
    #현재 노드 방문 처리
    visited[V] = 1
    print(V, end = ' ')
    #현재 노드와 연결된 다른 노드를 재귀적으로 방문

        if visited[i] == 0:
            dfs(li,i,visited)

    deq = deque([V])
    visited[V] = 0

        for i in li[V]:
            if visited[i] == 1:
                deq.append(i)
                visited[i] = 0
