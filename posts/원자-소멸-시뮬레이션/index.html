#include <iostream>
#include <vector>
#include <cstring>

using namespace std;

#define MAX_ARR_LEN 4001

int T, N, answer = 0;


typedef struct {
	int x;
	int y;
	int dir;
	int energy;
	bool islive;
}atom;

vector<atom> ainfo;

int arrmap[MAX_ARR_LEN][MAX_ARR_LEN] = { };

int dx[4] = { 0, 0, -1, 1 };
int dy[4] = { 1, -1, 0, 0 }; 
void input_func() {
	int x, y, dir, K;
	cin >> N;
	answer = 0;
	memset(arrmap, 0, sizeof(arrmap));
	ainfo.clear();
	for (int i = 0; i < N; i++) {
		cin >> x >> y >> dir >> K;
		y = (y + 1000) * 2;
		x = (x + 1000) * 2;
		ainfo.push_back({ x, y, dir, K, true });  
		arrmap[x][y] = 1;
	}
	

}

bool all_vanish_check_func() {

	for (int i = 0; i < ainfo.size(); i++) {
		if (ainfo[i].islive)
			return false;
	}

	return true;
}

bool edge_check_func(int x, int y) {

	return (x >= 0 && x < MAX_ARR_LEN && y >= 0 && y < MAX_ARR_LEN);
}

void simulate() {

		// ì›ì ì¢Œí‘œ ì—…ë°ì´íŠ¸

		for (int i = 0; i < ainfo.size(); i++) {
			if (!ainfo[i].islive)
				continue;
			int cx = ainfo[i].x;
			int cy = ainfo[i].y;
			
			int dir = ainfo[i].dir;
			int nx = cx + dx[dir];
			int ny = cy + dy[dir];
			
			if (edge_check_func(ny, nx)) {
				//arrmap[ny][nx] -= 1;
				arrmap[cx][cy] = 0;
				arrmap[nx][ny] += 1;
