격자 안에서 완전탐색 최고의 33위치

N * N 크기의 격자 정보가 주어집니다. 이때 해당 위치에 동전이 있다면 1, 없다면 0이 주어집니다. N * N 격자를 벗어나지 않도록 3 * 3 크기의 격자를 적절하게 잘 잡아서 해당 범위 안에 들어있는 동전의 개수를 최대로 하는 프로그램을 작성해보세요.

#include <cstdio>

using namespace std;

int N;
int arr[20][20] = { 0 };

int main() {
    // 여기에 코드를 작성해주세요.
    int max_cnt = 0;
    scanf("%d", &N);
    for(int i = 0; i < N; i++)
    {
        for(int j = 0; j < N; j++)
        {
            scanf("%d", &arr[i][j]);   
        }
    }

    
    for(int i = 0; i < N; i++)
    {
        for(int j = 0; j < N; j++)
        {
            int temp_cnt = 0;
            for(int y = i; y < i + 3; y++)
            {
                for(int x = j; x < j + 3; x++)
                {
                    if(arr[y][x] == 1)
                        temp_cnt++;

                }
            }
            if(max_cnt < temp_cnt)
                max_cnt = temp_cnt;
        }
    }
    printf("%d\n", max_cnt);

    return 0;
}
