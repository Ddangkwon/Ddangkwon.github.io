const int INF = 1e9;

int main() {
    int n, m; // ì •ì , ê°„ì„ 
    cin >> n >> m;
    vector<vector<pair<int,int>>> g(n); // (to, w) w in {0,1}
    for (int i = 0; i < m; ++i) {
        int u, v, w;
        cin >> u >> v >> w; // 0 ë˜ëŠ” 1
        g[u].push_back({v, w});
        g[v].push_back({u, w});
    }
    deque<int> dq;
    vector<int> dist(n, INF);
    int s = 0;
    dist[s] = 0;
    dq.push_front(s);
    while (!dq.empty()) {
        int u = dq.front(); dq.pop_front();
        for (auto [v, w] : g[u]) {
            if (dist[v] > dist[u] + w) {
                dist[v] = dist[u] + w;
                if (w == 0) dq.push_front(v);
                else dq.push_back(v);
            }
        }
    }
    for (int i = 0; i < n; ++i) cout << dist[i] << " ";
    cout << "\n";
    return 0;
}
