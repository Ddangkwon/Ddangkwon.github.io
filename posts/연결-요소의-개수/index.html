#include <iostream>
#include <cstdio>
#include <vector>
#include <queue>
using namespace std;

#define MAX_NUM 1001

vector<int> connected[MAX_NUM];
bool isvisited[MAX_NUM] = { false,  };
int N, M;

void bfs(int num)
{
	queue<int> q;
	 
	isvisited[num] = true;
	q.push(num);

	while (!q.empty())
	{
		int cur = q.front();
		q.pop();
		for (int i = 0; i < connected[cur].size(); i++)
		{
			int next = connected[cur][i];
			if (!isvisited[next])
			{
				isvisited[next] = true;
				q.push(next);
			}
		}
	}
}
int main()
{
	int a, b, answer = 0;
	scanf("%d %d", &N, &M);
	for (int i = 0; i < M; i++)
	{
		scanf("%d %d", &a, &b);
		connected[b].push_back(a);
		connected[a].push_back(b);
	}
	for (int i = 1; i <= N; i++)
	{
		if (!isvisited[i])
		{
			answer++;
			bfs(i); 
		}
	}
	printf("%d", answer);
	return 0;
}
