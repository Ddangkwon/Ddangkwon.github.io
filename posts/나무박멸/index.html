#include <iostream>
#include <cstdio>

using namespace std;


/*
첫 번째 줄에 격자의 크기 n, 박멸이 진행되는 년 수 m, 제초제의 확산 범위 k, 제초제가 남아있는 년 수 c

1. 인접한 네 개의 칸 중 나무가 있는 칸의 수만큼 나무가 성장합니다. 성장은 모든 나무에게 동시에 일어납니다

2. 기존에 있었던 나무들은 인접한 4개의 칸 중 벽, 다른 나무, 제초제 모두 없는 칸에 번식을 진행합니다. 
이때 각 칸의 나무 그루 수에서 총 번식이 가능한 칸의 개수만큼 나누어진 그루 수만큼 번식이 되며, 나눌 때 생기는 나머지는 버립니다. 
번식의 과정은 모든 나무에서 동시에 일어나게 됩니다.

3. 각 칸 중 제초제를 뿌렸을 때 나무가 가장 많이 박멸되는 칸에 제초제를 뿌립니다. 
제초제를 뿌릴 때 4개의 대각선 방향으로 k칸만큼 전파되게 됩니다. 
단 전파되는 도중 벽이 있거나 나무가 아얘 없는 칸이 있는 경우, 그 칸 까지는 제초제가 뿌려지며 그 이후의 칸으로는 제초제가 전파되지 않습니다. 
제초제가 뿌려진 칸에는 c년만큼 제초제가 남아있다가 c+1년째가 될 때 사라지게 됩니다. 
제초제가 뿌려진 곳에 다시 제초제가 뿌려지는 경우에는 새로 뿌려진 해로부터 다시 c년동안 제초제가 유지됩니다.

*/
int n, m, ks, c, answer = 0;

// 상하좌우 탐색용 
int b_dy[4] = { -1, 1, 0, 0 };
int b_dx[4] = { 0,  0, -1, 1 };

// 대각 탐색용
int k_dy[4] = { -1, -1, 1, 1};
int k_dx[4] = { -1, 1, 1, -1};

int tree_map[20][20] = { {0,}, };
int herb_map[20][20] = { {0, }, };
void input_func()
{
	//첫 번째 줄에 격자의 크기 n, 박멸이 진행되는 년 수 m, 제초제의 확산 범위 k, 제초제가 남아있는 년 수 c
	scanf("%d %d %d %d", &n, &m, &ks, &c);
	for (int i = 0; i < n; i++)
	{
		for (int j = 0; j < n; j++)
		{
			scanf("%d", &tree_map[i][j]);
		}
	}
}

//1. 인접한 네 개의 칸 중 나무가 있는 칸의 수만큼 나무가 성장합니다. 성장은 모든 나무에게 동시에 일어납니다
void growth()
{
	
	for (int i = 0; i < n; i++)
	{
		for (int j = 0; j < n; j++)
		{
			// 벽이거나 나무가 없는 경우
			if (tree_map[i][j] <= 0)
				continue;
			int cnt = 0;
			for (int k = 0; k < 4; k++)
			{
				int ny = i + b_dy[k];
				int nx = j + b_dx[k];
				if(ny < 0 || nx < 0 || ny >= n || nx >= n)
					continue;
				if (tree_map[ny][nx] > 0)
					cnt++;
			}
			tree_map[i][j] += cnt;
		}
	}

	return;
}


void breeding()
{
    // 번식 알고리즘의 경우 주변의 값에 영향을 주기 때문에 한 번에 적용하기 위해 temp 변수를 설정한다.
    int temp_tree[20][20] = 0;
	for (int i = 0; i < n; i++)
	{
		for (int j = 0; j < n; j++)
		{
			// 벽이거나 나무가 없는 경우
			
			if (tree_map[i][j] <= 0)
				continue;
			int cnt = 0;
			int startp = tree_map[i][j];
			for (int k = 0; k < 4; k++)
			{
				int ny = i + b_dy[k];
				int nx = j + b_dx[k];
				if (ny < 0 || nx < 0 || ny >= n || nx >= n)
					continue;
				if (herb_map[ny][nx] != 0)
					continue;
				// 빈공간인 경우
				if (tree_map[ny][nx] == 0)
					cnt++;
			}
			if (cnt != 0)
			{
				for (int k = 0; k < 4; k++)
				{
					int ny = i + b_dy[k];
					int nx = j + b_dx[k];
					if (ny < 0 || nx < 0 || ny >= n || nx >= n)
						continue;
					if (herb_map[ny][nx] != 0)
						continue;
					// 빈공간인 경우
					if (tree_map[ny][nx] == 0)
						temp_tree[ny][nx] += (startp / cnt);
				}
			}
			
		}
	}

    for (int i = 0; i < n; i++)
	{
		for (int j = 0; j < n; j++)
		{
            tree_map[i][j] += temp_tree[i][j];
        }
    }
	return;
}

void kill_tree()
{
	// 나무를 가장많이 죽일 수 있는 포인트를 찾는다.
	int max_y = 0, max_x = 0, max_kill = 0;
	for (int i = 0; i < n; i++)
	{
		for (int j = 0; j < n; j++)
		{
			
			if (tree_map[i][j] <= 0)
				continue;
			int killcnt = tree_map[i][j];
			for (int k = 0; k < 4; k++)
			{
                int ny = i;
                int nx = j;
				for (int l = 1; l <= ks; l++)
				{
					ny += k_dy[k];
					nx += k_dx[k];
					if (ny < 0 || nx < 0 || ny >= n || nx >= n)
						break;
					if (tree_map[ny][nx] <= 0)
						break;
					killcnt += tree_map[ny][nx];
				}
				
			}
			if (killcnt > max_kill)
			{
				max_kill = killcnt;
				max_y = i;
				max_x = j;
			}
		}
	}
	// 정답 업데이트(1년치)
	answer += max_kill;
	// 제초제 업데이트

	if (tree_map[max_y][max_x] > 0)
	{
		tree_map[max_y][max_x] = 0;
		herb_map[max_y][max_x] = c;

		for (int i = 0; i < 4; i++)
		{
            int ny = max_y;
            int nx = max_x;
			for (int j = 1; j <= ks; j++)
			{
				ny += k_dy[i];
				nx += k_dx[i];
				if (ny < 0 || nx < 0 || ny >= n || nx >= n)
					break;
				if (tree_map[ny][nx] < 0)
					break;
				// 빈공간인 경우 거기까지만 제초제 뿌리고 탈출
				if (tree_map[ny][nx] == 0)
				{
					herb_map[ny][nx] = c;
					break;
				}
				tree_map[ny][nx] = 0;
				herb_map[ny][nx] = c;
			}
		}
	}

}

//1년이 지남에 따라 제초제 1씩 감소
void vanish_herb()
{
	for(int i = 0; i < n; i++)
	{
		for (int j = 0; j < n; j++)
		{
			if (herb_map[i][j] > 0)
				herb_map[i][j]--;
		}
	}

}
int main()
{
	
	input_func();
	for (int i = 0; i < m; i++)
	{
		// 성장
		growth();
		// 번식
		breeding();
        // 제초제 지속기간 업데이트
		vanish_herb();
		// 박멸
		kill_tree();
		
	}
	printf("%d", answer);
	return 0;
}
