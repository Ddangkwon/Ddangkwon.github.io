#include <iostream>
#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>

#define LEN_1D_MAX 2500
#define LEN_2D_MAX 49
using namespace std;

int N, M, answer = 0;

int line_size = 0;
int line[LEN_1D_MAX] = { 0, };
int arr[LEN_2D_MAX][LEN_2D_MAX] = { 0, };

int dy[4] = { -1, 1, 0, 0 };
int dx[4] = { 0, 0, -1, 1 };
void destroy_by_rule(int d, int s)
{
	int cy = N / 2;
	int cx = N / 2;

	for (int i = 1; i <= s; i++)
	{
		int ny = cy + (dy[d] * i);
		int nx = cx + (dx[d] * i);
		arr[ny][nx] = 0;
	}
}

void convert_to_line()
{
	int cy = N / 2;
	int cx = N / 2;
	
	int ddy[4] = { 0, 1, 0, -1 };
	int ddx[4] = { -1, 0, 1, 0 };

	int dir = 0, loop_cnt = 2;

	line_size = 0;

	for (int move_r = 1; move_r < N; move_r++)
	{
		// ë§ˆì§€ë§‰ì˜ ê²½ìš°ì—ë§Œ loop cnt ê°€ 3ë²ˆìœ¼ë¡œ ëŠ˜ì–´ë‚œë‹¤
		if (move_r == N - 1)
		{
			loop_cnt = 3;
		}
		for (int i = 0; i < loop_cnt; i++)
		{
			for (int j = 0; j < move_r; j++)
			{
				cy += ddy[dir];
				cx += ddx[dir];
				if (arr[cy][cx] != 0)
				{
					line[line_size++] = arr[cy][cx];
				}
			}
			dir = (dir + 1) % 4;
		}

		

	}
	return;
}

int explosion()
{
	int point = 0;
	int cnt = 1;

	for (int i = 1; i < line_size; i++)
	{
		if (line[i - 1] == line[i])
		{
			++cnt;
		}
		else
		{
			if (cnt >= 4)
			{
				for (int j = 1; j <= cnt; j++)
				{
					point += line[i - j];
					line[i - j] = 0;
				}
			}
			cnt = 1;
		}
	}
	// ë§ˆì§€ë§‰ê¹Œì§€ ë™ì¼í•œ ìˆ«ìê°€ ë°˜ë³µë˜ê³  4ê°€ ë„˜ëŠ” ê²½ìš°ì— ëŒ€í•œ ì˜ˆì™¸ì²˜ë¦¬ë¥¼ í•´ì¤€ë‹¤

	{
		for (int j = 1; j <= cnt; j++)
		{
			point += line[line_size - j];
			line[line_size - j] = 0;
		}
	}

	answer += point;
	return point;
}

void remove_zero()
{
	int back_size = 0;
	int backup[2500] = { 0, };

	for (int i = 0; i < line_size; i++)
	{
		if (line[i] != 0)
		{
			backup[back_size++] = line[i];
		}
	}
	for (int i = 0; i < back_size; i++)
	{
		line[i] = backup[i];
	}
	line_size = back_size;

	return;
}

void change_by_rule()
{
	int back_size = 0;
	int backup[2500] = { 0, };
	int cnt = 1;
	for (int i = 1; i < line_size; i++)
	{
		if (line[i - 1] == line[i])
		{
			++cnt;
		}
		else
		{
			if (back_size < N * N - 2)

		}

	if (line_size > 0 && back_size < N * N - 2)
	{
		backup[back_size++] = cnt;
		backup[back_size++] = line[line_size - 1];
	}
	for (int i = 0; i < back_size; i++)
	{
		line[i] = backup[i];
	}
	line_size = back_size;
	return;
}

void convert_to_mat()
{
	int cy = N / 2;
	int cx = N / 2;

	int ddy[4] = { 0, 1, 0, -1 };
	int ddx[4] = { -1, 0, 1, 0 };

	int dir = 0, loop_cnt = 2;
	int cur = 0;


	for (int move_r = 1; move_r < N; move_r++)
	{

				{
					arr[cy][cx] = 0;
				}

		}

		

	}

	return;
}
