#define ARR_MAX_LEN 101
using namespace std;

int arr[ARR_MAX_LEN][ARR_MAX_LEN] = { {0, }, };
int temp[ARR_MAX_LEN][ARR_MAX_LEN] = { {0,}, };
int N, M, R;

void input_func()
{
    cin >> N >> M >> R;

    for (int i = 0; i < N; i++) {
        for (int j = 0; j < M; j++) {
            cin >> arr[i][j];
        }
    }
    
}

void print_answer_func(){
	for (int i = 0; i < N; i++) {
        for (int j = 0; j < M; j++) {
            cout << arr[i][j] << " ";
        }
		cout << "\n";
    }
}

void copy_arr()
{
    for (int i = 0; i < N; i++)
    {
        for (int j = 0; j < M; j++)
        {
            arr[i][j] = temp[i][j];
        }
    }
}

void func1(){

        for (int j = 0; j < M; j++)

        }
    }
    copy_arr();
}

void func2(){

void func5()

    int div_m = M / 2;
    // 1 ==> 2
    for (int i = 0; i < div_n; i++)
    {

            temp[i][j + div_m] = arr[i][j];
        }
    }
    // 2 ==> 3
    for (int i = 0; i < div_n; i++)
    {
        for (int j = 0; j < div_m; j++)
        {
            temp[i + div_n][j + div_m] = arr[i][j + div_m];

}
