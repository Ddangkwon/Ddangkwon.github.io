#include <iostream>
#include <cstdio>


using namespace std;

/*
초기 특수 영양제는 n x n 격자의 좌하단의 4개의 칸에 주어집니다.
1년동안 다음 단계를 거쳐 리브로수는 성장하게 됩니다.

1. 특수 영양제를 이동 규칙에 따라 이동시킵니다.

2. 특수 영양제를 이동 시킨 후 해당 땅에 특수 영양제를 투입합니다.

3. 특수 영양제를 투입한 리브로수의 대각선으로 인접한 방향에 높이가 1 이상인 리브로수가 있는 만큼 높이가 더 성장합니다. 대각선으로 인접한 방향이 격자를 벗어나는 경우에는 세지 않습니다.

4. 특수 영양제를 투입한 리브로수를 제외하고 높이가 2 이상인 리브로수는 높이 2를 베어서 잘라낸 리브로수로 특수 영양제를 사고, 해당 위치에 특수 영양제를 올려둡니다.

*/
int n, m, answer = 0;// 격자의 크기 n, 리브로수를 키우는 총 년 수 m
int tree_map[15][15] = { {0,}, };
int dir[100] = { 0, };
int p[100] = { 0, };
int dy[8] = { 0, -1, -1, -1,  0,  1, 1, 1};
int dx[8] = { 1,  1,  0, -1, -1, -1, 0, 1};
int diag_y[4] = {-1, -1, 1, 1};
int diag_x[4] = {-1, 1, 1, -1};


bool fert[15][15] = { {false,}, };
bool fert_temp[15][15] = { {false,}, }; // 영양제 업데이트 내용 임시 저장용
void input_func()
{
	scanf("%d %d", &n, &m);
	for (int i = 0; i < n; i++)
	{
		for (int j = 0; j < n; j++)
		{
			scanf("%d", &tree_map[i][j]);
		}
	}

	for (int i = 0; i < m; i++)
	{
		scanf("%d %d", &dir[i], &p[i]);
	}

	return;
}
void init_med()
{
	for (int i = n - 2; i < n; i++)
	{
		for (int j = 0; j < 2; j++)
		{
			fert[i][j] = true;
		}
	}

	return;
}

void move(int d, int p)
{
	// fert 업데이트 용 임시 변수 저장
	for (int i = 0; i < n; i++)
		for (int j = 0; j < n; j++)
			fert_temp[i][j] = false;
	// 이동 업데이트
	for (int i = 0; i < n; i++)
	{
		for (int j = 0; j < n; j++)
		{
			if (fert[i][j])
			{
				int ny = (i + dy[d] * p + n * p) % n;
				int nx = (j + dx[d] * p + n * p) % n;
				fert_temp[ny][nx] = true;
			}
			
		}
	}
	// 임시변수 실제 영양제 지도에 덮어쓰기
	for (int i = 0; i < n; i++)
		for (int j = 0; j < n; j++)
			fert[i][j] = fert_temp[i][j];

			
	return;
}

void insert()
{
	// 영양제가 있는 곳에 나무 높이 1 증가
	for (int i = 0; i < n; i++)
	{
		for (int j = 0; j < n; j++)
		{
			if (fert[i][j])
			{
				tree_map[i][j]++;
			}
		}
	}

	//영양제가 있는 땅에 대각 방향에 리브로수 개수 만큼 높이 증가
	for (int i = 0; i < n; i++)
	{
		for (int j = 0; j < n; j++)
		{
			if (fert[i][j])
			{	
				int cnt = 0;
				for (int k = 0; k < 4; k++)
				{
					int ny = i + diag_y[k];
					int nx = j + diag_x[k];
					if (nx < 0 || ny < 0 || nx >= n || ny >= n)
						continue;
					if (tree_map[ny][nx] >= 1)
						cnt++;
				}
				tree_map[i][j] += cnt;
			}
			
		}
	}

	

	return;
}

void update()
{
	// fert 업데이트 용 임시 변수 저장
	for (int i = 0; i < n; i++)
		for (int j = 0; j < n; j++)
			fert_temp[i][j] = false;

	// 나무 가지치기 및 영양제 업데이트
	for (int i = 0; i < n; i++)
	{
		for (int j = 0; j < n; j++)
		{
			if (tree_map[i][j] >= 2 && !fert[i][j])
			{
				tree_map[i][j] -= 2;
				fert_temp[i][j] = true;
			}
		}
	}
		


	// 임시변수 실제 영양제 지도에 덮어쓰기
	for (int i = 0; i < n; i++)
		for (int j = 0; j < n; j++)
			fert[i][j] = fert_temp[i][j];
	return;
}

void calc_answer()
{

	for (int i = 0; i < n; i++)
		for (int j = 0; j < n; j++)
			answer += tree_map[i][j];
}
int main()
{

	input_func();
	// 초기 영양제 위치
	init_med();
	 
	for (int i = 0; i < m; i++)
	{
		// 1. 영양제 이동
		move(dir[i]-1, p[i]);
		// 2. 영양제 투입, 3. 성장 결과 업데이트
		insert();


		// 4. 나무 가지치기 및 영양제 재 투입
		update();
	}
	calc_answer();
	printf("%d\n", answer);
	return 0;
}
