#include <iostream>
#include <map>
#include <string>

int main() {
    string text = "hello world";
    map<char, int> freq;

    for (char ch : text) {
        if (ch != ' ')
            freq[ch]++;
    }

    for (const auto& [c, count] : freq) {
        std::cout << c << ": " << count << "\n";
    }

    return 0;
}
