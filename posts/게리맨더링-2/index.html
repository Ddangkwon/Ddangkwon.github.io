#include <iostream>
#include <cstdio>
#include <cmath>
#include <vector>
#include <algorithm>
using namespace std;

int N, answer = 98765432;
int arr[21][21] = { {0, }, };

int sum = 0;
void input_func()
{
	scanf("%d", &N);
	for (int i = 1; i <= N; i++)
	{
		for (int j = 1; j <= N; j++)
		{
			scanf("%d", &arr[i][j]);
			sum += arr[i][j];
		}
	}
}

int diff(int x, int y, int d1, int d2)
{
	vector<int> population(5, 0);
	int temp_arr[21][21] = { {0,}, };
	temp_arr[x][y] = 5;
	//(x, y), (x + 1, y - 1), ..., (x + d1, y - d1)
	//(x + d2, y + d2), (x + d2 + 1, y + d2 - 1), ..., (x + d2 + d1, y + d2 - d1)
	for (int i = 1; i <= d1; i++)
	{
		temp_arr[x + i][y - i] = 5;
		temp_arr[x + d2 + i][y + d2 - i] = 5;
	}
	//(x, y), (x + 1, y + 1), ..., (x + d2, y + d2)
	//(x + d1, y - d1), (x + d1 + 1, y - d1 + 1), ... (x + d1 + d2, y - d1 + d2)
	for(int i = 1; i <= d2; i++)
	{
		temp_arr[x + i][y + i] = 5;
		temp_arr[x + d1 + i][y - d1 + i] = 5;
	}

	//1ë²ˆ ì„ ê±°êµ¬ : 1 â‰¤ r < x + d1, 1 â‰¤ c â‰¤ y
	
	for (int r = 1; r < x + d1; r++)
	{
		for (int c = 1; c <= y; c++)
		{
			if (temp_arr[r][c] == 5)
			{
				break;
			}
			else
			{
				population[0] += arr[r][c];
			}
		}
	}
	//2ë²ˆ ì„ ê±°êµ¬ : 1 â‰¤ r â‰¤ x + d2, y < c â‰¤ N
	for (int r = 1; r <= x + d2; r++)
	{
		for (int c = N; c > y; c--)
		{
			if (temp_arr[r][c] == 5)
			{
				break;
			}
			else
			{
				population[1] += arr[r][c];
			}
		}
	}
	//3ë²ˆ ì„ ê±°êµ¬ : x + d1 â‰¤ r â‰¤ N, 1 â‰¤ c < y - d1 + d2
	for (int r = x + d1; r <= N; r++)
	{
		for (int c = 1 ; c < y - d1 + d2; c++)
		{
			if (temp_arr[r][c] == 5)
			{
				break;
			}
			else
			{
				population[2] += arr[r][c];
			}
		}
	}
	//4ë²ˆ ì„ ê±°êµ¬ : x + d2 < r â‰¤ N, y - d1 + d2 â‰¤ c â‰¤ N
	for (int r = x + d2 + 1; r <= N; r++)
	{
		for (int c = N; c >= y - d1 + d2; c--)
		{
			if (temp_arr[r][c] == 5)
			{
				break;
			}
			else
			{
				population[3] += arr[r][c];
			}
		}
	}
	//for (int r = 1; r <= N; r++)
	//{
	//	for (int c = 1; c <= N; c++)
	//	{
	//		if (temp_arr[r][c] == 5)
	//		{
	//			population[4] += arr[r][c];
	//		}
	//	}
	//}
	population[4] = sum - population[3] - population[2] - population[1] - population[0];
	sort(population.begin(), population.end());

	return population[4] - population[0];
}

void sector()
{
	//(d1, d2 â‰¥ 1 => ë¬´ì¡°ê±´ ì°¸, 1 â‰¤ x( => ë¬´ì¡°ê±´ ì°¸) < x+d1+d2 â‰¤ N, 1 â‰¤ y-d1 < y < y+d2 â‰¤ N)
	for (int x = 1; x <= N; x++)
	{
		for (int y = 1; y <= N; y++)
		{
			for (int d1 = 1; d1 <= N; d1++)
			{
				for (int d2 = 0; d2 <= N ;d2++)
				{
					if (x + d1 + d2 > N || 1 > y - d1 || y + d2 > N)
						continue;
					answer = min(answer, diff(x, y, d1, d2));

				}
			}
		}
	}
}
