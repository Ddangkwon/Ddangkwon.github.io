마법사 상어와 토네이도

문제를 총 2가지로 나눠보면,

  1. 토네이도의 이동 경로 설정
  2. 비율 별로 모래를 흩날리게 하기

이 두가지이고 먼저 1번의 경우 패턴을 보면 방향전환을 총 2번한 후 이동 거리가 1씩 증가하는 것을 알 수있다. 이러한 나선형 이동 구조를 바탕으로 어떤 크기든 최종적으로 종착점은 0,0이다.(종료 조건)

2번의 경우 기존 dx, dy 이동 배열 뿐만 아니라 방향 별로 모래가 흩날리게 되는 비율을 설정하는 배열 t_dx, t_dy를 추가로 설정하고 문제에 사용한다.

#include <iostream>
#include <cstdio>
#include <cmath>
#include <cstring>
using namespace std;

float s_rat[9] = { 0.01, 0.01, 0.07, 0.07, 0.1, 0.1, 0.02, 0.02, 0.05 };

//서남동북 순
int dy[4] = { 0, 1, 0, -1 };
int dx[4] = { -1, 0, 1, 0 };

int t_dy[4][10] = {
				{-1, 1, -1, 1, -1, 1, -2, 2, 0, 0}, // 서
				{-1, -1, 0, 0, 1, 1, 0, 0, 2, 1}, // 남
				{1, -1, 1, -1, 1, -1, 2, -2, 0, 0}, // 동
				{1, 1, 0, 0, -1, -1, 0, 0, -2, -1} // 북
				};
int t_dx[4][10] = {
				{1, 1, 0, 0, -1, -1, 0, 0, -2, -1}, // 서
				{-1, 1, -1, 1, -1, 1,-2, 2, 0, 0}, // 남
				{-1, -1, 0, 0, 1, 1, 0, 0, 2, 1}, // 동
				{1, -1, 1, -1, 1, -1, 2, -2, 0, 0} // 북
				};

int N, answer = 0;
int arr[500][500] = { { 0,}, };
/*
토네이도는 한 번에 한 칸 이동한다
토네이도 회전을 보면 이동방향을 두번 틀 때마다 이동 거리가 1씩 증가한다.
결과적으로 크기가 몇이든 0,0 점이 종착점이 된다.
*/

void input_func()
{
	scanf("%d", &N);
	for (int i = 0; i < N; i++)
	{
		for (int j = 0; j < N; j++)
		{
			scanf("%d", &arr[i][j]);
		}
	}
}

void sand_checker()
{
	int cx = N / 2;
	int cy = N / 2;
	
	
	int dir = 0; // 0 : 서, 1 : 남, 2 : 동, 3 : 북
	int dist = 1; // 이동 간격
	int cnt = 0; // 두 번 체크 변수
	int sandtemp = 0;
	//탈출조건 허리케인이 종착지 0, 0에 도달 했을 때
	while(true)
	{	
		cnt++;

		for (int i = 0; i < dist; i++)
		{

			int nx = cx + dx[dir];
			int ny = cy + dy[dir];
			int sandtemp = arr[ny][nx];
			for (int j = 0; j < 9; j++)
			{
				
				int t_nx = nx + t_dx[dir][j];
				int t_ny = ny + t_dy[dir][j];
				int tpointsand = arr[ny][nx] * s_rat[j];
				sandtemp -= tpointsand;
				if (t_nx < 0 || t_ny < 0 || t_nx >= N || t_ny >= N)
					answer += tpointsand;
				else
					arr[t_ny][t_nx] += tpointsand;

			}
			
			int t_nx = nx + t_dx[dir][9];
			int t_ny = ny + t_dy[dir][9];

			if (t_nx < 0 || t_ny < 0 || t_nx >= N || t_ny >= N)
				answer += sandtemp;
			else
				arr[t_ny][t_nx] += sandtemp;
			
			arr[ny][nx] = 0; // 해당 지역 모래는 모두 사라진다.
			
			if (nx == 0 && ny == 0)
				return;
			cx = nx;
			cy = ny;
			
			
		}

		if (cnt == 2)
		{
			cnt = 0;
			dist++;

		}
		dir = (dir + 1) % 4; // 반시계 방향으로 90도 방향회전
		
	}

	return;
}
int main(int argc, char** argv)
{
	input_func();
	sand_checker();
	printf("%d\n", answer);
	return 0;//정상종료시 반드시 0을 리턴해야합니다.
}
