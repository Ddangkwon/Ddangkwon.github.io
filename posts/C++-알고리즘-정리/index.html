void dfs_iterative(int start) {
    stack<int> s;
    s.push(start);

    while (!s.empty()) {
        int node = s.top(); s.pop();
        if (visited[node]) continue;

        visited[node] = true;
        cout << node << ' ';

        for (int i = graph[node].size() - 1; i >= 0; --i) {
            int next = graph[node][i];
            if (!visited[next]) {
                s.push(next);
            }
        }
    }
}

int n;
bool isvisited[10]; // ìˆ«ì ì‚¬ìš© ì—¬ë¶€ ì²´í¬

void dfs(vector<int>& perm, int depth) {
    if (depth == n) {
        for (int num : perm) {
            cout << num << " ";
        }
        cout << "\n";
        return;
    }

    for (int i = 1; i <= n; i++) {
        if (!isvisited[i]) {
            isvisited[i] = true;
            perm.push_back(i);
            dfs(perm, depth + 1);
            perm.pop_back();
            isvisited[i] = false;
        }
    }
}

int main() {
    cin >> n;
    vector<int> perm;
    dfs(perm, 0);
}

    sort(groups + 1, groups + group_id + 1);

    printf("%d\n", group_id);
    for (int i = 1; i <= group_id; i++) {
        printf("%d\n", groups[i]);
    }
}

using namespace std;

int weight[5] = {16, 8, 4, 2, 1};

int main(){
    int a, b, c, d, e, N, answer = 0;
    vector<int> arr;
    cin >> a >> b >> c >> d >> e >> N;

    arr.push_back(e);
    arr.push_back(d);
    arr.push_back(c);
    arr.push_back(b);
    arr.push_back(a);


    for(int i = 0; i < 5; i++){
        int used = N / weight[i];
        if(used > arr[i])
            used = arr[i];
        N -= (used * weight[i]);
        answer += used;

    }
    if(N == 0)
        cout << answer << '\n';
    else
        cout << "impossible" << '\n';

    return 0;
}

#define MAX_TABLE (10007)

unsigned long hash(const char* str)
{
    unsigned long hash = 5381;
    int c;

    while (c = *str++)
    {
        hash = (((hash << 5) + hash) + c) % MAX_TABLE;
    }

    return hash % MAX_TABLE;
}
