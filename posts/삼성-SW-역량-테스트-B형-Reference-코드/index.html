unsigned long hash(const char *str)
{
	unsigned long hash = 5381;
	int c;

	while (c = *str++)
	{
		hash = (((hash << 5) + hash) + c) % MAX_TABLE;
	}

	return hash % MAX_TABLE;
}

#define ARR_MAX 110

int b[ARR_MAX];
int marr[ARR_MAX];
void merge(int* arr, int start, int end)
{
	int mid, i, j, k;

	mid = (start + end) >> 1;
	i = k = start;
	j = mid + 1;
	while (i <= mid && j <= end)
	{
		if (arr[i] < arr[j])
		{
			marr[k++] = arr[i++];
		}
		else
		{
			marr[k++] = arr[j++];
		}
	}
	while (i <= mid)
	{
		marr[k++] = arr[i++];
	}
		
	while (j <= end)
	{
		marr[k++] = arr[j++];
	}

	for (int i = start; i <= end; i++)
	{
		arr[i] = marr[i];
	}

}
void sort(int* arr, int start, int end)
{
	if (start >= end)
		return;
	int mid = (start + end) >> 1;
	sort(arr, start, mid);
	sort(arr, mid + 1, end);
	merge(arr, start, end);
	
}

long long d[100];

// н”јліҙлӮҳм№ҳ н•ЁмҲҳлҘј мһ¬к·Җн•ЁмҲҳлЎң кө¬нҳ„(нғ‘лӢӨмҡҙ лӢӨмқҙлӮҳлҜ№ н”„лЎңк·ёлһҳл°Қ)
int fibo(int x) {
	// мў…лЈҢмЎ°кұҙ(1нҳ№мқҖ 2мқјл•Ң 1л°ҳнҷҳ)
    if(x==1 || x==2){
    	return 1;
    }
    // мқҙлҜё кі„мӮ°н•ңм Ғ мһҲлҠ” л¬ём ңлқјл©ҙ к·ёлҢҖлЎң л°ҳнҷҳ

    	return d[x];

    return d[x] = fibo(x-1) + fibo(x-2);
}

лӢӨмқҙлӮҳлҜ№ н”„лЎңк·ёлһҳл°Қ VS л¶„н•  м •ліө

кіөнҶөм җ : лӘЁл‘җ мөңм Ғ л¶Җл¶„ кө¬мЎ°лҘј к°Җм§Ҳ л•Ң мӮ¬мҡ© к°ҖлҠҘ

(нҒ° л¬ём ңлҘј мһ‘мқҖ л¬ём ңлЎң лӮҳлҲҢ мҲҳ мһҲмңјл©° мһ‘мқҖ л¬ём ңмқҳ лӢөмқ„ лӘЁм•„м„ң нҒ° л¬ём ңлҘј н•ҙкІ°н•  мҲҳ мһҲлҠ” мғҒнҷ©)

м°Ёмқҙм җ : л¶Җл¶„ л¬ём ң м •ліө.

(лӢӨмқҙлӮҳлҜ№ н”„лЎңк·ёлһҳл°Қ л¬ём ңм—җм„ңлҠ” к°Ғ л¶Җл¶„ л¬ём ңл“Өмқҙ м„ңлЎң мҳҒн–Ҙмқ„ лҜём№ҳл©° л¶Җл¶„ л¬ём ңк°Җ мӨ‘ліөлҗҳкі ,

л¶„н•  м •ліө л¬ём ңм—җм„ңлҠ” лҸҷмқјн•ң л¶Җл¶„ л¬ём ңк°Җ л°ҳліөм ҒмңјлЎң кі„мӮ°лҗҳм§Җ м•ҠлҠ”лӢӨ.)

л¶„н•  м •ліөмқҳ лҢҖн‘ң мӮ¬лЎҖмқё нҖө м •л ¬м—җм„ңлҠ” н•ң лІҲ кё°мӨҖ мӣҗмҶҢк°Җ мһҗлҰ¬лҘј ліҖкІҪн•ҙм„ң мһҗлҰ¬лҘј мһЎмңјл©ҙ, к·ё кё°мӨҖ мӣҗмҶҢмқҳ мң„м№ҳк°Җ л°”лҖҢм§Җ м•ҠлҠ”лӢӨ.

лҳҗн•ң л¶„н•  мқҙнӣ„м—җ н•ҙлӢ№ н”јлІ—мқ„ лӢӨмӢң мІҳлҰ¬н•ҳлҠ” л¶Җл¶„м—җм„ң лӢӨмӢң нҳём¶ңн•ҳм§Җ м•ҠлҠ”лӢӨ.

лҸҷм Ғ кі„нҡҚлІ•мқҖ мқҙлҜё н•ҙкІ°н•ң л¬ём ңлҘј л°ҳліөм ҒмңјлЎң н•ҙкІ°н•ҳм—¬ л№„нҡЁмңЁм Ғмқё л¬ём ңм—җм„ң мң мҡ©н•  кІғмқҙлӢӨ.

#include <stdio.h>

#define ARR_MAX 110

int heap[ARR_MAX];
int hn;
int N;
int pop(int* heap, int& hn)
{
	register int tmp, ret;
	//1) popмқҳ returnк°’ heap[1]мқ„ кё°м–өн•ҙл‘”лӢӨ.
	ret = heap[1];
	//2) heapмқҳ л°°м—ҙм—җм„ң к°ҖмһҘ лҒқм—җ к°’, мҰү heap[hn]мқ„ heap[1]лЎң ліҖкІҪн•ңлӢӨ.
	heap[1] = heap[hn];
	//3) heap[hn]мқҖ мөңм•…мқҳ мҡ°м„ мҲңмң„к°’(мөңмҶҢнһҷмқҙлҜҖлЎң 0x7fff0000 м •лҸ„мқҳ нҒ° к°’)мңјлЎң ліҖкІҪн•ҙл‘”лӢӨ.
	//4) heapмқҙ 1к°ң мӨ„м—ҲмңјлҜҖлЎң hn мқ„ 1 к°җмҶҢ мӢңнӮЁлӢӨ.
	heap[hn--] = 0x7fff0000;

	for (register int i = 1; i * 2 <= hn;)
	{
		if (heap[i] < heap[i * 2] && heap[i] < heap[i * 2 + 1]) break;
		else if (heap[i * 2] < heap[i * 2 + 1])
		{
			tmp = heap[i * 2];
			heap[i * 2] = heap[i];
			heap[i] = tmp;

			i = i * 2;
		}
		else
		{
			tmp = heap[i * 2 + 1];
			heap[i * 2 + 1] = heap[i];
			heap[i] = tmp;

			i = i * 2 + 1;
		}
	}

	return ret;
}

void push(int* heap, int& hn, int x)
{
	register int tmp;
	//лӘЁл“  iм—җ лҢҖн•ҙм„ң мһҗмӢқ nodeлҠ” i * 2, i * 2 + 1мқҙ лҗҳкі ,
	//л¶ҖлӘЁ nodeлҠ” i / 2лЎң н•ҳл©ҙ лҗңлӢӨ.

	heap[++hn] = x;
	for (register int i = hn; i > 1; i /= 2)
	{
		// мһҗмӢқ л…ёл“ңк°Җ лҚ” нҒ¬л©ҙ pass
		if (heap[i / 2] <= heap[i]) 
			break;
		// м•„лӢҲл©ҙ мЎ°мғҒ <-> л¶ҖлӘЁ <-> мһҗмӢқ л…ёл“ңк°„м—җ көҗнҷҳмқ„ л°ҳліөн•ҙм„ң мҲҳн–үн•ңлӢӨ. (л¶ҖлӘЁл…ёл“ңк°Җ мһҗмӢқл…ёл“ңліҙлӢӨ мһ‘мқ„л•Ңк№Ңм§Җ)
		// л¶ҖлӘЁ л…ёл“ңмҷҖ мһҗмӢқ л…ёл“ңмҷҖмқҳ көҗнҷҳ
		tmp = heap[i / 2];
		heap[i / 2] = heap[i];
		heap[i] = tmp;
	}
}
